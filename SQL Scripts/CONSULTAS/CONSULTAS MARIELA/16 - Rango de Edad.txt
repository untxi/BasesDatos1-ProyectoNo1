CREATE OR REPLACE FUNCTION get_rango_edad(pedad1 NUMBER, pedad2 NUMBER)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
	      vEdad number;
	      vFecha number;
       BEGIN 

         SELECT PERSONA_ID FROM PERSONA
         WHERE 
		pedad1 <=
		
		(SELECT extract(YEAR FROM SYSDATE) INTO vFecha FROM DUAL;
    		 SELECT extract(YEAR FROM NACIMIENTO) INTO vEdad FROM PERSONA;)
		
		<=pedad2;

       RETURN vPersona;
       END;
