CREATE OR REPLACE FUNCTION get_rango_altura(paltura1_id NUMBER, paltura2_id NUMBER)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

         SELECT PERSONA_ID FROM PERSONA
         WHERE (PERSONA.ALTURA_ID >= paltura1_id) AND (PERSONA.ALTURA_ID <= paltura2_id);

       RETURN vPersona;
       END;
