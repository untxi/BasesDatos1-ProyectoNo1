CREATE OR REPLACE FUNCTION get_Idioma(pIdioma_id NUMBER)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

	 SELECT persona_id FROM IDIOMA_X_PERSONA
         WHERE IDIOMA_X_PERSONA.IDIOMA_ID = pIdioma_id;

       RETURN vPersona;

       END;
