CREATE OR REPLACE FUNCTION get_Deporte1()
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

	 SELECT COUNT(persona_id) FROM DEPORTE_X_PERSONA
         WHERE DEPORTE_X_PERSONA.PERSONA_ID > 1;

       RETURN vPersona;

       END;
       
--Buscar todas las personas con almenos 1 deporte
