CREATE OR REPLACE FUNCTION get_Mismo_Idioma(pMi_Id)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

	 SELECT P.persona_id, IXP.persona_id  
	 FROM PERSONA P INNER JOIN IDIOMA_X_PERSONA IXP
         WHERE P.IDIOMA_ID = pMi_Id and P.IDIOMA_ID = IXP.IDIOMA_ID;

       RETURN vPersona;

       END;
       
--Buscar todas las personas con MI MISMO idioma
