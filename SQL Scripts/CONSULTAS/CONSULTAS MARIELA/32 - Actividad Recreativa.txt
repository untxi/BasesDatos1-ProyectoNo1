CREATE OR REPLACE FUNCTION get_ActividadR(pActividadR_id NUMBER)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

	 SELECT persona_id FROM ACTIVIDAD_X_PERSONA
         WHERE ACTIVIDAD_X_PERSONA.ACTIVIDAD_RECREATIVA_ID = pActividadR_id;

       RETURN vPersona;

       END;
