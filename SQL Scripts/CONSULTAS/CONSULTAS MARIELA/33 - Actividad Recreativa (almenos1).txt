CREATE OR REPLACE FUNCTION get_Actividad1()
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

	 SELECT COUNT(persona_id) FROM ACTIVIDAD_X_PERSONA
         WHERE ACTIVIDAD_X_PERSONA.PERSONA_ID > 1;

       RETURN vPersona;

       END;
       
--Buscar todas las personas con almenos 1 actividad
