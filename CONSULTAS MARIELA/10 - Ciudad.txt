CREATE OR REPLACE FUNCTION get_Ciudad(pCiudad_id number)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

         SELECT persona_id, nombre, primer_apellido FROM PERSONA
         WHERE PERSONA.CIUDAD_ID = pCiudad_id;

       RETURN vPersona;

       END;
