CREATE OR REPLACE FUNCTION get_Deporte(pDeporte_id NUMBER)
       RETURN NUMBER
       AS
              vPersona NUMBER(9);
       BEGIN 

	 SELECT persona_id FROM DEPORTE_X_PERSONA
         WHERE DEPORTE_X_PERSONA.DEPORTE_ID = pDeporte_id;

       RETURN vPersona;

       END;
